Основные объекты системы:

// Удаленный вызов процедур

// Структура объекта
RPC({onSuccess:fn, onError:fn})

	// вызов удаленного метода
	method({
		name: str, 
		params: {}|[],
		onSuccess: fn, 
		onError: fn,
		deferred: false | true | Number // отложенное выполнение, по умолчанию false, если цифра - миллисекунды до отправки
	})
	
	// получение данных
	get({
		param: {}|[], 
		onSuccess: fn, 
		onError: fn
	})
	
	// отправка данных
	set({
		param: {}|[], 
		onSuccess: fn, 
		onError: fn
	})
	
	// синхронизация
	sync()
